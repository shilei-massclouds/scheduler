7/// A task wrapper for the [`RRScheduler`].
8///
9/// It add a time slice counter to use in round-robin scheduling.
10pub struct RRTask<T, const MAX_TIME_SLICE: usize> {
11    inner: T,
12    time_slice: AtomicIsize,
13}
14
15impl<T, const S: usize> RRTask<T, S> {
16    /// Creates a new [`RRTask`] from the inner task struct.
17    pub const fn new(inner: T) -> Self {
18        Self {
19            inner,
20            time_slice: AtomicIsize::new(S as isize),
21        }
22    }
23
24    fn time_slice(&self) -> isize {
25        self.time_slice.load(Ordering::Acquire)
26    }
27
28    fn reset_time_slice(&self) {
29        self.time_slice.store(S as isize, Ordering::Release);
30    }
31
32    /// Returns a reference to the inner task struct.
33    pub const fn inner(&self) -> &T {
34        &self.inner
35    }
36}
37
38impl<T, const S: usize> Deref for RRTask<T, S> {
39    type Target = T;
40    #[inline]
41    fn deref(&self) -> &Self::Target {
42        &self.inner
43    }
44}

46/// A simple [Round-Robin] (RR) preemptive scheduler.
47///
48/// It's very similar to the [`FifoScheduler`], but every task has a time slice
49/// counter that is decremented each time a timer tick occurs. When the current
50/// task's time slice counter reaches zero, the task is preempted and needs to
51/// be rescheduled.
52///
53/// Unlike [`FifoScheduler`], it uses [`VecDeque`] as the ready queue. So it may
54/// take O(n) time to remove a task from the ready queue.
55///
56/// [Round-Robin]: https://en.wikipedia.org/wiki/Round-robin_scheduling
57/// [`FifoScheduler`]: crate::FifoScheduler
58pub struct RRScheduler<T, const MAX_TIME_SLICE: usize> {
59    ready_queue: VecDeque<Arc<RRTask<T, MAX_TIME_SLICE>>>,
60}
61
62impl<T, const S: usize> RRScheduler<T, S> {
63    /// Creates a new empty [`RRScheduler`].
64    pub const fn new() -> Self {
65        Self {
66            ready_queue: VecDeque::new(),
67        }
68    }
69    /// get the name of scheduler
70    pub fn scheduler_name() -> &'static str {
71        "Round-robin"
72    }
73}
74
75impl<T, const S: usize> BaseScheduler for RRScheduler<T, S> {
76    type SchedItem = Arc<RRTask<T, S>>;
77
78    fn init(&mut self) {}
79
80    fn add_task(&mut self, task: Self::SchedItem) {
81        self.ready_queue.push_back(task);
82    }
83
84    fn remove_task(&mut self, task: &Self::SchedItem) -> Option<Self::SchedItem> {
85        // TODO: more efficient
86        self.ready_queue
87            .iter()
88            .position(|t| Arc::ptr_eq(t, task))
89            .and_then(|idx| self.ready_queue.remove(idx))
90    }
91
92    fn pick_next_task(&mut self) -> Option<Self::SchedItem> {
93        self.ready_queue.pop_front()
94    }
95
96    fn put_prev_task(&mut self, prev: Self::SchedItem, preempt: bool) {
97        if prev.time_slice() > 0 && preempt {
98            self.ready_queue.push_front(prev)
99        } else {
100            prev.reset_time_slice();
101            self.ready_queue.push_back(prev)
102        }
103    }
104
105    fn task_tick(&mut self, current: &Self::SchedItem) -> bool {
106        let old_slice = current.time_slice.fetch_sub(1, Ordering::Release);
107        old_slice <= 1
108    }
109
110    fn set_priority(&mut self, _task: &Self::SchedItem, _prio: isize) -> bool {
111        false
112    }
113}
